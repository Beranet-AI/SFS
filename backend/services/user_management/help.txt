1.برای زاه اندازی جنگو ابتدا باید در محیط توسعه ای که کا میکنم ابتدا فریم ورک جنگو را نصب کنیم
2.سپس دستور زیر را در همان محیط وارد کنیم
pip install djangorestframework djangorestframework-simplejwt
3.نصب درایور دیتابیس
pip install psycopg2-binary
4.python manage.py startapp farm
5.python manage.py startapp devices
6.ثبت appها در settings.py
user_management/config/settings.py
INSTALLED_APPS = [
    # ...
    "rest_framework",
    "rest_framework_simplejwt",

    "farm",
    "devices",
]
7.نگاشت Farm / Barn / Zone به مدل‌های Django
در فایل:
user_management/farm/models.py

8.نگاشت Device / SensorType / Sensor به مدل‌های Django
در فایل:
user_management/devices/models.py

9.cd SmartFarmSystemEnterprise/backend/services/user_management
python manage.py makemigrations farm devices

10.SmartFarmSystemEnterprise/backend/services/user_management/config/settings.py
DATABASES = {
    "default": {
        "ENGINE": "django.db.backends.sqlite3",
        "NAME": BASE_DIR / "db.sqlite3",
    }
}

11.python manage.py migrate
